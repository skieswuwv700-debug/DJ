local function creatTweenPart()
    local part = Instance.new("Part")
    part.Size = Vector3.new(1, 1, 1)
    part.Anchored = true
    part.CanCollide = false
    part.CFrame = player.Character.HumanoidRootPart.CFrame
    part.Transparency = 1
    part.Name = "TweenPart"
    part.Parent = workspace
    return part
end

local function tp(pos)
    isTweening = false
    local currentCharacter = player.Character or character
    local rootPart = currentCharacter and currentCharacter:FindFirstChild("HumanoidRootPart")
    if rootPart then
        hrp = rootPart
        rootPart.CFrame = pos
    end
end

local function tween(targetCFrame, _cancel)
    local TweenPart = workspace:FindFirstChild("TweenPart") or creatTweenPart()
    TweenPart.CFrame = player.Character.HumanoidRootPart.CFrame
    if getgenv().startTween then getgenv().startTween:Cancel() end
    local distance = (targetCFrame.Position - TweenPart.Position).Magnitude
    if distance < 1.5 then
        TweenPart.CFrame = targetCFrame
        isTweening = true
        return
    end
    local TweenService = game:GetService("TweenService")
    local tweenInfo = TweenInfo.new(distance / 300, Enum.EasingStyle.Linear)
    isTweening = true
    getgenv().startTween = TweenService:Create(TweenPart, tweenInfo, {CFrame = targetCFrame})
    getgenv().startTween:Play()
    getgenv().startTween.Completed:Wait()
    if _cancel then isTweening = false end
end

local function equipTool(typeTool)
    local success, equipped = pcall(function()
        local backpackFolder = player:FindFirstChild("Backpack")
        if not backpackFolder then return false end
        for _, v in pairs(backpackFolder:GetChildren()) do
            if v:IsA("Tool") and (v.ToolTip == typeTool or string.find(v.Name, typeTool, 1, true)) then
                humanoid:EquipTool(v)
                return true
            end
        end
        return false
    end)
    return success and equipped
end

local function findPhysicalFruit()
    local backpack = player:FindFirstChild("Backpack")
    if backpack then
        for _, tool in ipairs(backpack:GetChildren()) do
            if tool:IsA("Tool") and string.find(tool.Name, "Fruit", 1, true) then
                return tool
            end
        end
    end
    local char = player.Character or character
    if char then
        for _, tool in ipairs(char:GetChildren()) do
            if tool:IsA("Tool") and string.find(tool.Name, "Fruit", 1, true) then
                return tool
            end
        end
    end
    return nil
end

local function dropFruitByEatRemote(fruit)
    if not fruit then return false end
    local char = player.Character or character
    local backpack = player.Backpack
    if not char then return false end
    for _, child in ipairs(char:GetChildren()) do
        if child:IsA("Tool") then child.Parent = backpack end
    end
    task.wait(0.2)
    fruit.Parent = char
    local eatRemote = fruit:FindFirstChild("EatRemote")
    if not eatRemote then
        fruit.Parent = backpack
        warn("ไม่พบ EatRemote ในผลไม้: " .. fruit.Name)
        return false
    end
    local success, err = pcall(function()
        eatRemote:InvokeServer("Drop")
    end)
    if not success then
        warn("Drop fruit failed: " .. tostring(err))
        fruit.Parent = backpack
        return false
    end
    return true
end

local function getAllFruit()
    local itemData = require(game:GetService("ReplicatedStorage").Economy.ItemId.RawSource)
    local allFruit = {}
    for _, data in pairs(itemData) do
        if data.Id and data.Id.Type == "Fruit" then
            local itemName, itemId = data.Id.StorageKey, data.Id.ItemId
            if not allFruit[itemName] then allFruit[itemName] = {} end
            table.insert(allFruit[itemName], itemId)
        end
    end
    return allFruit
end

local function checkItem(targetList)
    local InventoryController = require(ReplicatedStorage.Controllers.UI.Inventory)
    local foundIds = {}
    for _, item in pairs(InventoryController:GetTiles()) do
        for name, target in pairs(targetList) do
            if item.ItemId == target[1] then
                if not foundIds[name] then
                    table.insert(foundIds, name)
                    foundIds[name] = true
                end
            end
        end
    end
    return foundIds
end

local function getUserTradingFruitData(_displaname)
    local filePath = folderName .. "/" .. _displaname .. ".json"
    if not isfile(filePath) then return {} end
    local rawData = readfile(filePath)
    local success, userData = pcall(function() return HttpService:JSONDecode(rawData) end)
    if not success or not userData or not userData.inventory then return {} end
    local result = {}
    for _, fruit in ipairs(userData.inventory) do
        local cost = fruitsCost[fruit]
        if cost then result[fruit] = cost end
    end
    return result
end

local function calculateTotalValue(_me, _them)
    local meData = getUserTradingFruitData(_me)
    local themData = getUserTradingFruitData(_them)
    local meList, themList = {}, {}
    for fruit, cost in pairs(meData) do table.insert(meList, {name=fruit,cost=cost}) end
    for fruit, cost in pairs(themData) do table.insert(themList, {name=fruit,cost=cost}) end
    local themNames, meNames = {}, {}
    for _, t in ipairs(themList) do themNames[t.name] = true end
    for _, m in ipairs(meList) do meNames[m.name] = true end
    local meFiltered, themFiltered = {}, {}
    for _, m in ipairs(meList) do if not themNames[m.name] then table.insert(meFiltered, m) end end
    for _, t in ipairs(themList) do if not meNames[t.name] then table.insert(themFiltered, t) end end
    table.sort(meFiltered, function(a,b) return a.cost < b.cost end)
    table.sort(themFiltered, function(a,b) return a.cost < b.cost end)
    local MAX_DIFF_PCT = 0.40
    local function diffPct(a,b)
        if a==0 or b==0 then return 1 end
        return math.abs(a-b)/math.max(a,b)
    end
    local nMe = math.min(#meFiltered,12)
    local nThem = math.min(#themFiltered,12)
    local themMap = {}
    for tMask=1,(2^nThem)-1 do
        local subset,total,count={},0,0
        for i=1,nThem do
            if bit32.band(tMask,2^(i-1))~=0 then
                table.insert(subset,themFiltered[i])
                total=total+themFiltered[i].cost
                count=count+1
            end
        end
        if not themMap[total] or themMap[total].count>count then
            themMap[total]={fruits=subset,count=count}
        end
    end
    local bestResult,bestCount,bestDiff=nil,math.huge,-1
    for mMask=1,(2^nMe)-1 do
        local meSubset,meTotal,meCount={},0,0
        for i=1,nMe do
            if bit32.band(mMask,2^(i-1))~=0 then
                table.insert(meSubset,meFiltered[i])
                meTotal=meTotal+meFiltered[i].cost
                meCount=meCount+1
            end
        end
        for themTotal,themEntry in pairs(themMap) do
            local pct=diffPct(meTotal,themTotal)
            local count=meCount+themEntry.count
            if pct<=MAX_DIFF_PCT then
                local better=false
                if count<bestCount then better=true
                elseif count==bestCount and pct>bestDiff then better=true end
                if better then
                    bestCount=count; bestDiff=pct
                    bestResult={me=meSubset,them=themEntry.fruits,meTotal=meTotal,themTotal=themTotal,diffAbs=math.abs(meTotal-themTotal),diffPct=pct*100}
                end
            end
        end
    end
    if not bestResult then return nil end
    local fruitIdsByName = getAllFruit()
    local result = {}
    for _, fruit in ipairs(bestResult.me) do
        local fruitIds = fruitIdsByName[fruit.name]
        if fruitIds and fruitIds[1] then table.insert(result,fruitIds[1])
        else warn("Missing fruit item id for "..tostring(fruit.name)) end
    end
    return result
end

local questArgs = {{ NPC="Dojo Trainer", Command="RequestQuest" }}
local dacro_module = ReplicatedStorage:WaitForChild("Modules"):WaitForChild("Net"):WaitForChild("RF/InteractDragonQuest")

local function sucsessTradeQuest()
    local item = {["OrangeBelt"]={213}}
    if checkItem(item)["OrangeBelt"] then return true end
    return false
end

local function sentQuest()
    dacro_module:InvokeServer({ NPC="Dojo Trainer", Command="ClaimQuest" })
end

local function helperTrade()
    if player:GetAttribute("ExactLocation") ~= "Mansion" then
        tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
    elseif isReadyToTrade() then
        local aloowedFruits = calculateTotalValue(player.DisplayName, getUserNameTrade())
        if aloowedFruits then
            for i, v in pairs(aloowedFruits) do
                warn(i, v); task.wait(0.5); addFruit(v)
            end
            repeat task.wait(0.3); acceptTrade() until not isSit()
        else
            if isSit() then humanoid.Sit = false end
            tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
            warn("none fruit to add"); task.wait(10)
        end
        savePlayerData(player.DisplayName, checkItem(getAllFruit()))
    elseif not isSit() then
        tp(tradeChairPos[1]); task.wait(1)
    end
end

local function normalTrade()
    local Progress = nil
    local function getQP()
        tween(CFrame.new(5868.453125, 1207.7784423828125, 870.8200073242188)); task.wait(1)
        local res = dacro_module:InvokeServer(unpack(questArgs))
        if not res["Timeout"] then
            task.wait(2); Progress = dacro_module:InvokeServer(unpack(questArgs))["Quest"]["Progress"]
        else Progress = -1 end
    end
    if Progress == 1 then
        tween(CFrame.new(5868.453125, 1207.7784423828125, 870.8200073242188)); task.wait(1); sentQuest()
    elseif player:GetAttribute("ExactLocation") ~= "Mansion" then
        tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
    elseif isReadyToTrade() then
        local aloowedFruits = calculateTotalValue(player.DisplayName, getUserNameTrade())
        if aloowedFruits then
            for i, v in pairs(aloowedFruits) do
                warn(i, v); task.wait(0.5); addFruit(v)
            end
            repeat task.wait(0.3); acceptTrade() until not isSit()
            getQP(); task.wait(1)
        else
            if isSit() then humanoid.Sit = false end
            tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
            warn("none fruit to add"); task.wait(10)
        end
        savePlayerData(player.DisplayName, checkItem(getAllFruit()))
    elseif not isSit() then
        tp(getTradeChairPos()); task.wait(1)
    end
end

if getgenv().Configs.Mode == "trade" then
    warn("in loop ...")
    savePlayerData(player.DisplayName, checkItem(getAllFruit()))
    task.wait(3)
    while true do task.wait()
        if sucsessTradeQuest() then helperTrade() else normalTrade() end
    end
end

local Progress

local function requestQuestProgress()
    local success, response = pcall(function()
        return dacro_module:InvokeServer(unpack(questArgs))
    end)
    if not success or not response then return nil end
    if response["Timeout"] then Progress = -1; return Progress end
    local quest = response["Quest"]
    if quest and quest["Progress"] ~= nil then
        Progress = quest["Progress"]; return Progress
    end
    return nil
end

local function getQuestProgress()
    if not requestQuestProgress() and not Progress then
        tween(CFrame.new(5868.453125, 1207.7784423828125, 870.8200073242188), true)
        task.wait(2); requestQuestProgress()
    else
        task.wait(2); requestQuestProgress()
    end
    task.wait(1)
    if Progress == nil then Progress = 0 end
    return Progress
end

local function getCurrentHrp()
    local currentCharacter = player.Character or character
    if not currentCharacter then return nil end
    local rootPart = currentCharacter:FindFirstChild("HumanoidRootPart")
    if rootPart then hrp = rootPart; return rootPart end
    return nil
end

local DROP_DISTANCE = 2
local DROP_OFFSET_DONE = CFrame.new(-20, 0, 0)

local function moveToDropPos(offset)
    local currentHrp = getCurrentHrp()
    if not currentHrp then return false end
    local target = getDropFruitPos()
    if offset then target = target * offset end
    if (currentHrp.Position - target.Position).Magnitude > DROP_DISTANCE then
        tp(target); task.wait(0.2)
    end
    return true
end

local function shouldRandomFruit()
    return findPhysicalFruit() == nil
end

local function hasFruit()
    return findPhysicalFruit() ~= nil
end

local function dropFruitAtDropPos()
    if not moveToDropPos() then return false end
    local fruit = findPhysicalFruit()
    if not fruit then return false end
    local char = player.Character or character
    if not char then return false end
    if fruit.Parent ~= char then
        fruit.Parent = char
        task.wait(0.4)
    end
    task.wait(0.2)
    local dropped = dropFruitByEatRemote(fruit)
    if dropped then task.wait(0.5) end
    return dropped
end

local function sucsessDropQuest()
    local item = {["BlueBelt"]={211}}
    if checkItem(item)["BlueBelt"] then return true end
    return false
end

local function dropFruit()
    if not getCurrentHrp() then return end

    if isHelperPlayer() then
        if shouldRandomFruit() then
            if random() then task.wait(1) end
            return
        end

        if dropFruitAtDropPos() then return end

        if not Progress then
            getQuestProgress(); return
        end

        moveToDropPos()

        if sucsessDropQuest() then
            sentQuest()
            Progress = nil
            task.wait(0.5)
            return
        end

        if not sucsessDropQuest() and getQuestProgress() >= 1 then
            sentQuest(); task.wait(0.5)
        end
    else
        if sucsessDropQuest() then
            if hasFruit() then
                dropFruitAtDropPos()
            else
                moveToDropPos(DROP_OFFSET_DONE)
            end
            return
        end

        if dropFruitAtDropPos() then return end

        if not Progress then
            getQuestProgress(); return
        end

        if not sucsessDropQuest() and getQuestProgress() >= 1 then
            sentQuest(); task.wait(0.5); return
        end

        moveToDropPos()
    end
end

if getgenv().Configs.Mode == "drop" then
    while true do task.wait()
        dropFruit()
    end
end
