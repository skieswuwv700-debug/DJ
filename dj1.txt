local function creatTweenPart()
    local part = Instance.new("Part")
    part.Size = Vector3.new(1, 1, 1)
    part.Anchored = true
    part.CanCollide = false
    part.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
    part.Transparency = 1
    part.Name = "TweenPart"
    part.Parent = workspace
    return part
end

local function tp(pos)
    isTweening = false
    character.HumanoidRootPart.CFrame = pos
end

local function tween(targetCFrame, _cancel)
    local TweenPart = workspace:FindFirstChild("TweenPart") or creatTweenPart()
    TweenPart.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame

    if getgenv().startTween then
        getgenv().startTween:Cancel()
    end

    local distance = (targetCFrame.Position - TweenPart.Position).Magnitude
    
    if distance < 1.5 then
        TweenPart.CFrame = targetCFrame
        isTweening = true
        return
    end

    local TweenService = game:GetService("TweenService")
    local tweenInfo = TweenInfo.new(distance / 350, Enum.EasingStyle.Linear)
    isTweening = true

    getgenv().startTween = TweenService:Create(TweenPart, tweenInfo, {CFrame = targetCFrame})
    getgenv().startTween:Play()
    getgenv().startTween.Completed:Wait()

    if _cancel then isTweening = false end
end

local function getUserTradingFruitData(_displaname)
    local filePath = folderName .. "/" .. _displaname .. ".json"
    if not isfile(filePath) then return {} end
    local rawData = readfile(filePath)
    local success, userData = pcall(function() return HttpService:JSONDecode(rawData) end)
    if not success or not userData or not userData.inventory then return {} end
    local result = {}
    for _, fruit in ipairs(userData.inventory) do
        local cost = fruitsCost[fruit]
        if cost then result[fruit] = cost end
    end
    return result
end

local function calculateTotalValue(_me, _them)
    local meData   = getUserTradingFruitData(_me)
    local themData = getUserTradingFruitData(_them)
    local meList, themList = {}, {}
    for fruit, cost in pairs(meData)   do table.insert(meList,   { name = fruit, cost = cost }) end
    for fruit, cost in pairs(themData) do table.insert(themList, { name = fruit, cost = cost }) end
    local themNames, meNames = {}, {}
    for _, t in ipairs(themList) do themNames[t.name] = true end
    for _, m in ipairs(meList)   do meNames[m.name]   = true end
    local meFiltered, themFiltered = {}, {}
    for _, m in ipairs(meList)   do if not themNames[m.name] then table.insert(meFiltered,   m) end end
    for _, t in ipairs(themList) do if not meNames[t.name]   then table.insert(themFiltered, t) end end
    table.sort(meFiltered,   function(a, b) return a.cost < b.cost end)
    table.sort(themFiltered, function(a, b) return a.cost < b.cost end)
    local MAX_DIFF_PCT = 0.40
    local function diffPct(a, b)
        if a == 0 or b == 0 then return 1 end
        return math.abs(a - b) / math.max(a, b)
    end
    local nMe   = math.min(#meFiltered,   12)
    local nThem = math.min(#themFiltered, 12)
    local themMap = {}
    for tMask = 1, (2 ^ nThem) - 1 do
        local subset, total, count = {}, 0, 0
        for i = 1, nThem do
            if bit32.band(tMask, 2 ^ (i - 1)) ~= 0 then
                table.insert(subset, themFiltered[i])
                total = total + themFiltered[i].cost
                count = count + 1
            end
        end
        if not themMap[total] or themMap[total].count > count then
            themMap[total] = { fruits = subset, count = count }
        end
    end
    local bestResult, bestCount, bestDiff = nil, math.huge, -1
    for mMask = 1, (2 ^ nMe) - 1 do
        local meSubset, meTotal, meCount = {}, 0, 0
        for i = 1, nMe do
            if bit32.band(mMask, 2 ^ (i - 1)) ~= 0 then
                table.insert(meSubset, meFiltered[i])
                meTotal = meTotal + meFiltered[i].cost
                meCount = meCount + 1
            end
        end
        for themTotal, themEntry in pairs(themMap) do
            local pct   = diffPct(meTotal, themTotal)
            local count = meCount + themEntry.count
            if pct <= MAX_DIFF_PCT then
                local better = (count < bestCount) or (count == bestCount and pct > bestDiff)
                if better then
                    bestCount  = count
                    bestDiff   = pct
                    bestResult = { me = meSubset, them = themEntry.fruits, meTotal = meTotal, themTotal = themTotal }
                end
            end
        end
    end
    if not bestResult then return nil end
    local fruitIdsByName = getAllFruit()
    local result = {}
    for _, fruit in ipairs(bestResult.me) do
        local fruitIds = fruitIdsByName[fruit.name]
        if fruitIds and fruitIds[1] then
            table.insert(result, fruitIds[1])
        else
            warn("Missing fruit item id for " .. tostring(fruit.name))
        end
    end
    return result
end

local TradeProgress = { ["Timeout"] = nil, ["Progress"] = nil }

local function getTradeQuestProgress()
    tween(CFrame.new(5868.453125, 1207.7784423828125, 870.8200073242188))
    task.wait(1)
    local ok, result = pcall(function() return dacro_module:InvokeServer(unpack(questArgs)) end)
    if ok and type(result) == "table" and result["Quest"] then
        TradeProgress["Timeout"] = false
        TradeProgress["Progress"] = result["Quest"]["Progress"]
    else
        TradeProgress["Timeout"] = true
        TradeProgress["Progress"] = -1
    end
end

local function sucsessTradeQuest()
    return checkItem({["OrangeBelt"] = {213}})["OrangeBelt"] ~= nil
end

local function helperTrade()
    if player:GetAttribute("ExactLocation") ~= "Mansion" then
        tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
    elseif isReadyToTrade() then
        local aloowedFruits = calculateTotalValue(player.DisplayName, getUserNameTrade())
        if aloowedFruits then
            for i, v in pairs(aloowedFruits) do
                warn(i, v)
                task.wait(0.5)
                addFruit(v)
            end
            repeat task.wait(0.3) acceptTrade() until not isSit()
        else
            if isSit() then humanoid.Sit = false end
            tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
            warn("none fruit to add")
            task.wait(10)
        end
        savePlayerData(player.DisplayName, checkItem(getAllFruit()))
    elseif not isSit() then
        tp(tradeChairPos[1])
        task.wait(1)
    end
end

local function normalTrade()
    if TradeProgress["Progress"] == 1 then
        tween(CFrame.new(5868.453125, 1207.7784423828125, 870.8200073242188))
        task.wait(1)
        sentQuest()
    elseif player:GetAttribute("ExactLocation") ~= "Mansion" then
        tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
    elseif isReadyToTrade() then
        local aloowedFruits = calculateTotalValue(player.DisplayName, getUserNameTrade())
        if aloowedFruits then
            for i, v in pairs(aloowedFruits) do
                warn(i, v)
                task.wait(0.5)
                addFruit(v)
            end
            repeat task.wait(0.3) acceptTrade() until not isSit()
            getTradeQuestProgress()
            task.wait(1)
        else
            if isSit() then humanoid.Sit = false end
            tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
            warn("none fruit to add")
            task.wait(10)
        end
        savePlayerData(player.DisplayName, checkItem(getAllFruit()))
    elseif not isSit() then
        tp(getTradeChairPos())
        task.wait(1)
    end
end

local function mainTradeFruit()
    if TradeProgress["Progress"] == nil or TradeProgress["Timeout"] == nil then
        getTradeQuestProgress()
    elseif sucsessTradeQuest() or TradeProgress["Timeout"] then
        warn("Helper Trade")
        helperTrade()
    else
        normalTrade()
    end
end

    for _, v in pairs(character:GetChildren()) do
        if v:IsA("Tool") and string.find(v.Name, "Fruit") then return v.Name end

    for _, v in pairs(player:FindFirstChild("Backpack"):GetChildren()) do
        if string.find(v.Name, "Fruit") then return v.Name end
    end
    return nil
end

local function sucsessDropQuest()
    return checkItem({["BlueBelt"] = {211}})["BlueBelt"] ~= nil
end

local function dropFruit()
    local pos = getDropFruitPos()
    local dist = (hrp.Position - pos.Position).Magnitude

    -- ซื้อ fruit ถ้าไม่มีเลย
    if not isHasFruitInBP() and not isPlayerHoldFruit() then
        warn("No fruit, buying...")
        buyRandom()
        task.wait(1)
        return
    end

    -- equip จาก BP ก่อนถ้ายังไม่ได้ถือ
    if isHasFruitInBP() and not isPlayerHoldFruit() then
        warn("Equipping fruit...")
        equipTool(isHasFruitInBP())
        task.wait(0.3)
        return
    end

    -- ไปจุด drop
    if dist > 2 then
        warn("Teleporting to drop pos...")
        tp(pos)
        task.wait(0.3)
        return
    end

    -- drop fruit
    if isPlayerHoldFruit() then
        warn("Dropping: " .. tostring(isPlayerHoldFruit()))
        task.wait(0.5)
        drop(isPlayerHoldFruit())
        task.wait(0.5)
        return
    end

    -- check / claim quest
    if DropProgress == nil then
        warn("Checking drop quest...")
        getDropQuestProgress()
        return
    end

    if sucsessDropQuest() then
        warn("Drop quest done (BlueBelt)!")
        task.wait(2)
        return
    end

    if DropProgress >= 1 then
        warn("Claiming drop quest...")
        sentQuest()
        DropProgress = nil
        task.wait(1)
    else
        -- refresh progress
        getDropQuestProgress()
    end
end

local function calculateTotalValue(_me, _them)
    local meData   = getUserTradingFruitData(_me)
    local themData = getUserTradingFruitData(_them)

    local meList, themList = {}, {}
    for fruit, cost in pairs(meData)   do table.insert(meList,   { name = fruit, cost = cost }) end
    for fruit, cost in pairs(themData) do table.insert(themList, { name = fruit, cost = cost }) end

    local themNames, meNames = {}, {}
    for _, t in ipairs(themList) do themNames[t.name] = true end
    for _, m in ipairs(meList)   do meNames[m.name]   = true end

    local meFiltered, themFiltered = {}, {}
    for _, m in ipairs(meList)   do if not themNames[m.name] then table.insert(meFiltered,   m) end end
    for _, t in ipairs(themList) do if not meNames[t.name]   then table.insert(themFiltered, t) end end

    table.sort(meFiltered,   function(a, b) return a.cost < b.cost end)
    table.sort(themFiltered, function(a, b) return a.cost < b.cost end)

    local MAX_DIFF_PCT = 0.40

    local function diffPct(a, b)
        if a == 0 or b == 0 then return 1 end
        return math.abs(a - b) / math.max(a, b)
    end

    local nMe   = math.min(#meFiltered,   12)
    local nThem = math.min(#themFiltered, 12)

    local themMap = {}
    for tMask = 1, (2 ^ nThem) - 1 do
        local subset, total, count = {}, 0, 0
        for i = 1, nThem do
            if bit32.band(tMask, 2 ^ (i - 1)) ~= 0 then
                table.insert(subset, themFiltered[i])
                total = total + themFiltered[i].cost
                count = count + 1
            end
        end
        if not themMap[total] or themMap[total].count > count then
            themMap[total] = { fruits = subset, count = count }
        end
    end

    local bestResult = nil
    local bestCount  = math.huge
    local bestDiff   = -1

    for mMask = 1, (2 ^ nMe) - 1 do
        local meSubset, meTotal, meCount = {}, 0, 0
        for i = 1, nMe do
            if bit32.band(mMask, 2 ^ (i - 1)) ~= 0 then
                table.insert(meSubset, meFiltered[i])
                meTotal = meTotal + meFiltered[i].cost
                meCount = meCount + 1
            end
        end

        for themTotal, themEntry in pairs(themMap) do
            local pct   = diffPct(meTotal, themTotal)
            local count = meCount + themEntry.count

            if pct <= MAX_DIFF_PCT then
                local better = false
                if count < bestCount then
                    better = true
                elseif count == bestCount and pct > bestDiff then
                    better = true
                end

                if better then
                    bestCount  = count
                    bestDiff   = pct
                    bestResult = {
                        me        = meSubset,
                        them      = themEntry.fruits,
                        meTotal   = meTotal,
                        themTotal = themTotal,
                        diffAbs   = math.abs(meTotal - themTotal),
                        diffPct   = pct * 100,
                    }
                end
            end
        end
    end

    if not bestResult then return nil end

    local fruitIdsByName = getAllFruit()
    local result = {}
    for _, fruit in ipairs(bestResult.me) do
        local fruitIds = fruitIdsByName[fruit.name]
        if fruitIds and fruitIds[1] then
            table.insert(result, fruitIds[1])
        else
            warn("Missing fruit item id for " .. tostring(fruit.name))
        end
    end
    return result
end

local function helperTrade()
    if player:GetAttribute("ExactLocation") ~= "Mansion" then
        tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
    elseif isReadyToTrade() then
        local aloowedFruits = calculateTotalValue(player.DisplayName, getUserNameTrade())

        if aloowedFruits then
            for i, v in pairs(aloowedFruits) do
                warn(i, v)
                task.wait(0.5)
                addFruit(v)
            end

            repeat task.wait(0.3)
                acceptTrade()
            until not isSit()
        else
            if isSit() then humanoid.Sit = false end
            tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
            warn("none fruit to add")
            task.wait(10)

    elseif not isSit() then
        tp(tradeChairPos[1])
        task.wait(1)
    end
end

-- ฟังก์ชัน dropFruit แก้ใหม่ ไม่ยืนนิ่งแล้ว
local function dropFruit()
    local pos = getDropFruitPos()
    local distToPos = (hrp.Position - pos.Position).Magnitude

    -- ซื้อ fruit ก่อนถ้าไม่มีเลย
    if shouldRandomFruit() then
        warn("No fruit, buying...")
        random()
        task.wait(1)
        return
    end

    -- ถ้ามีใน BP ให้ถือออกมาก่อน
    if isHasFruitInBP() then
        warn("Equipping fruit from backpack...")
        equipTool(isHasFruitInBP())
        task.wait(0.3)
        return
    end

    -- เดินทางไปจุด drop ถ้ายังไม่ถึง
    if distToPos > 2 then
        warn("Teleporting to drop position...")
        tp(pos)
        task.wait(0.3)
        return
    end

    -- drop fruit ถ้าถือมาแล้ว
    if isPlayerHoldFruit() then
        warn("Dropping fruit: " .. tostring(shouldDropFruit()))
        task.wait(0.5)
        drop(shouldDropFruit())
        task.wait(0.5)
        return
    end

    -- check/claim quest
    if not DropProgress then
        warn("Checking quest progress...")
        getDropQuestProgress()
        return
    end

    if sucsessDropQuest() then
        warn("Quest already done (BlueBelt), idling...")
        task.wait(2)
        return
    end

    local currentProgress = getDropQuestProgress()
    if currentProgress and currentProgress >= 1 then
        warn("Claiming quest...")
        sentQuest()
        DropProgress = nil
        task.wait(1)
    end
end

local function creatTweenPart()
    local part = Instance.new("Part")
    part.Size = Vector3.new(1, 1, 1)
    part.Anchored = true
    part.CanCollide = false
    part.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
    part.Transparency = 1
    part.Name = "TweenPart"
    part.Parent = workspace
    return part
end

local function tp(pos)
    isTweening = false
    character.HumanoidRootPart.CFrame = pos
end

local function tween(targetCFrame, _cancel)
    local TweenPart = workspace:FindFirstChild("TweenPart") or creatTweenPart()
    TweenPart.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame

    if getgenv().startTween then
        getgenv().startTween:Cancel()
    end

    local distance = (targetCFrame.Position - TweenPart.Position).Magnitude
    
    if distance < 1.5 then
        TweenPart.CFrame = targetCFrame
        isTweening = true
        return
    end

    local TweenService = game:GetService("TweenService")
    local tweenInfo = TweenInfo.new(distance / 350, Enum.EasingStyle.Linear)
    
    isTweening = true

    getgenv().startTween = TweenService:Create(TweenPart, tweenInfo, {CFrame = targetCFrame})
    getgenv().startTween:Play()
    getgenv().startTween.Completed:Wait()

    if _cancel then isTweening = false end
end

        end
    end)
end

local function getAllFruit()
    local itemData = require(game:GetService("ReplicatedStorage").Economy.ItemId.RawSource)
    local allFruit = {}

    for _, data in pairs(itemData) do
        if data.Id and data.Id.Type == "Fruit" then
            local itemName, itemId = data.Id.StorageKey, data.Id.ItemId
            if not allFruit[itemName] then allFruit[itemName] = {} end
            table.insert(allFruit[itemName], itemId)
        end
    end
    return allFruit

local function checkItem(targetList)
    local InventoryController = require(ReplicatedStorage.Controllers.UI.Inventory)
    local ItemReplication = require(ReplicatedStorage.Util.ItemReplication)

    local foundIds = {}

    for _, item in pairs(InventoryController:GetTiles()) do
        for name, target in pairs(targetList) do
            if item.ItemId == target[1] then
                if not foundIds[name] then
                    table.insert(foundIds, name)
                    foundIds[name] = true
                end
            end
        end
    end
    return foundIds
end

    if type(result) == "table" and result["Quest"] then
        Progress["Timeout"] = false
        Progress["Progress"] = result["Quest"]["Progress"]
    elseif type(result) == "table" and result["Timeout"] then
        Progress["Timeout"] = true
        Progress["Progress"] = -1
    else
        Progress["Timeout"] = true
        Progress["Progress"] = -1
    end
end

local function sucsessTradeQuest()
    local item = {["OrangeBelt"] = {213}}
    if checkItem(item)["OrangeBelt"] then
        return true
    end
    return false
end

local function sentQuest()
    local args = {
        {
            NPC = "Dojo Trainer",
            Command = "ClaimQuest"
        }
    }
    game:GetService("ReplicatedStorage"):WaitForChild("Modules"):WaitForChild("Net"):WaitForChild("RF/InteractDragonQuest"):InvokeServer(unpack(args))
end

local function helperTrade()
    if player:GetAttribute("ExactLocation") ~= "Mansion" then
        tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
    elseif isReadyToTrade() then
        local aloowedFruits = calculateTotalValue(player.DisplayName, getUserNameTrade())

        if aloowedFruits then
            for i, v in pairs(aloowedFruits) do
                warn(i, v)
                task.wait(0.5)
                addFruit(v)
            end

            repeat task.wait(0.3)
                acceptTrade()
            until not isSit()
        else
            if isSit() then humanoid.Sit = false end
            tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
            warn("none fruit to add")
            task.wait(10)
        end

        savePlayerData(player.DisplayName, checkItem(getAllFruit()))
    elseif not isSit() then
        tp(tradeChairPos[1])
        task.wait(1)
    end
end

local function normalTrade()
    if Progress["Progress"] == 1 then
        tween(CFrame.new(5868.453125, 1207.7784423828125, 870.8200073242188))
        task.wait(1)
        sentQuest()
    elseif player:GetAttribute("ExactLocation") ~= "Mansion" then
        tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
    elseif isReadyToTrade() then
        local aloowedFruits = calculateTotalValue(player.DisplayName, getUserNameTrade())

        if aloowedFruits then
            for i, v in pairs(aloowedFruits) do
                warn(i, v)
                task.wait(0.5)
                addFruit(v)
            end

            repeat task.wait(0.3)
                acceptTrade()
            until not isSit()

            getQuestProgress()
            task.wait(1)
        else
            if isSit() then humanoid.Sit = false end
            tween(workspace._WorldOrigin.Locations.Mansion.CFrame * CFrame.new(0,100,0), true)
            warn("none fruit to add")
            task.wait(10)
        end

        savePlayerData(player.DisplayName, checkItem(getAllFruit()))
    elseif not isSit() then
        tp(getTradeChairPos())
        task.wait(1)
    end
end

local function mainTradeFruit()
    if Progress["Progress"] == nil or Progress["Timeout"] == nil then
        getQuestProgress()
    elseif sucsessTradeQuest() or Progress["Timeout"] then
        warn("Helper Trade")
        helperTrade()
    elseif not sucsessTradeQuest() then
        normalTrade()
    end
end

local function random()
    ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("CommF_"):InvokeServer("Cousin", "Buy", "DLCBoxData")
end

local DropProgress = nil

local function getDropQuestProgress()
    local result = dacro_module:InvokeServer(unpack(questArgs))

    if type(result) == "table" and result["Quest"] then
        DropProgress = result["Quest"]["Progress"]
    else
        DropProgress = nil
    end

    task.wait(1)
    return DropProgress
end

local dropFruitPos = {
    ["Helper"] = CFrame.new(5847.72266, 1208.31042, 879.862549, -0.622320533, 7.22706872e-08, 0.782762527, -3.3727261e-08, 1, -1.19141951e-07, -0.782762527, -1.00544916e-07, -0.622320533),
    ["Skibidi67"] = CFrame.new(5844.11279, 1208.31042, 882.85321, 0.650864184, 3.9195438e-08, -0.759194136, -9.16891736e-08, 1, -2.69782916e-08, 0.759194136, 8.71690915e-08, 0.650864184)
}

local function getDropFruitPos()
    if table.find(getgenv().Configs["Helper"], player.Name) then return dropFruitPos["Helper"] end
    return dropFruitPos["Skibidi67"]
end

local function isPlayerHoldFruit()
    for i, v in pairs(character:GetChildren()) do
        if v:IsA("Tool") and string.find(v.Name, "Fruit") then
            return v.Name
        end
    end
    return nil
end

local function isHasFruitInBP()
    for i, v in pairs(player:FindFirstChild("Backpack"):GetChildren()) do
        if string.find(v.Name, "Fruit") then
            return v.Name
        end
    end
    return nil
end

local function shouldDropFruit()
    local fruitName = isPlayerHoldFruit()
    if fruitName then
        return fruitName
    end
end

local function shouldRandomFruit()
    local shouldRandom = not isHasFruitInBP() and not isPlayerHoldFruit()
    return shouldRandom
end

local function sucsessDropQuest()
    local item = {["BlueBelt"] = {211}}
    if checkItem(item)["BlueBelt"] then
        return true
    end
    return false
end

local function dropFruit()
    if table.find(getgenv().Configs["Helper"], player.Name) then
        if not DropProgress then
            getDropQuestProgress()
        elseif not sucsessDropQuest() and getDropQuestProgress() >= 1 then
            sentQuest()
        elseif shouldRandomFruit() then
            random()
            task.wait(1)
        elseif (hrp.Position - getDropFruitPos().Position).Magnitude > 2 then
            tp(getDropFruitPos())
        elseif isPlayerHoldFruit() then
            tp(getDropFruitPos())
            task.wait(0.5)
            drop(shouldDropFruit())
        elseif isHasFruitInBP() then
            equipTool(isHasFruitInBP())
        end
    else
        if sucsessDropQuest() then
            tp(getDropFruitPos() * CFrame.new(-20, 0, 0))
            if isPlayerHoldFruit() then
                tp(getDropFruitPos())
                task.wait(1)
                drop(shouldDropFruit())
            end
            return
        elseif not DropProgress then
            getDropQuestProgress()
        elseif getDropQuestProgress() >= 1 then
            sentQuest()
        elseif (hrp.Position - getDropFruitPos().Position).Magnitude > 2 then
            tp(getDropFruitPos())
        elseif isPlayerHoldFruit() then
            tp(getDropFruitPos())
            task.wait(0.5)
            drop(shouldDropFruit())
        end
    end
end

if getgenv().Configs["Mode"] == "drop" then
    while true do task.wait()
        dropFruit()
    end
end
